<script>
let isPlaying = false;

function showWish(){
  const name = document.getElementById('nameInput').value.trim();
  if(!name){ alert("Please enter your name!"); return; }

  document.getElementById('nameDisplay').innerHTML =
    "🎈 Wishing you a wonderful birthday, <b>" + name + "</b>! 🎁";
  document.querySelector('.cake').style.display = 'block';
  document.getElementById('message').style.display = 'block';
  document.getElementById('slideshow').style.display = 'block';
  createConfetti();
  playMusic();
}

function createConfetti(){
  for(let i=0;i<80;i++){
    const c = document.createElement('div');
    c.classList.add('confetti');
    c.style.backgroundColor = randomColor();
    c.style.left = Math.random()*100+'vw';
    c.style.animationDuration = (Math.random()*3+2)+'s';
    c.style.width = c.style.height = (Math.random()*8+4)+'px';
    document.body.appendChild(c);
    setTimeout(()=>c.remove(),5000);
  }
}

function randomColor(){
  const colors=['#ff4e50','#f9d423','#4ef0ff','#9d50bb','#50ff85'];
  return colors[Math.floor(Math.random()*colors.length)];
}

function playMusic(){
  const audio = document.getElementById('bdaySong');
  const toggleBtn = document.getElementById('musicToggle');
  audio.loop = true;
  audio.volume = 0;
  audio.play().then(()=>{
    fadeInMusic(audio);
    isPlaying = true;
    toggleBtn.textContent = "🔊";
  }).catch(()=>{
    document.body.addEventListener('click',()=>{
      audio.play();
      fadeInMusic(audio);
      isPlaying = true;
      toggleBtn.textContent = "🔊";
    },{once:true});
    alert("🔈 Tap anywhere to start the Happy Birthday song!");
  });
}

function fadeInMusic(audio){
  let vol = 0;
  const fade = setInterval(()=>{
    if(vol < 1){
      vol += 0.02;
      audio.volume = vol;
    } else {
      clearInterval(fade);
    }
  },100);
}

document.getElementById('musicToggle').addEventListener('click',()=>{
  const audio = document.getElementById('bdaySong');
  const btn = document.getElementById('musicToggle');
  if(isPlaying){
    audio.pause();
    btn.textContent = "🔇";
    isPlaying = false;
  } else {
    audio.play();
    btn.textContent = "🔊";
    isPlaying = true;
  }
});
</script>
